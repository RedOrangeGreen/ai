/* MainScreen */
@Composable
fun MainScreen(mainViewModel: MainViewModel = viewModel()) {
    when (mainViewModel.currentScreen) {
        is Screen.Play -> PlayScreen(mainViewModel)
        is Screen.Settings -> SettingsScreen(mainViewModel)
        is Screen.About -> AboutScreen(mainViewModel)
    }
}

/* PlayScreen */
@Composable
fun PlayScreen(mainViewModel: MainViewModel) {
    val selectedChannel = remember { mutableStateOf(null) }
    var selectedVideoSource by remember { mutableStateOf(VideoSource.Video1) }

    LaunchedEffect(mainViewModel.currentPlaylist) {
        withContext(Dispatchers.IO) {
            mainViewModel.channels.clear() // Clear existing channels
            mainViewModel.channels.addAll(getChannels(mainViewModel.currentPlaylist))
        }
    }

    Column(
        modifier = Modifier.fillMaxSize(),
        verticalArrangement = Arrangement.Top,
        horizontalAlignment = Alignment.CenterHorizontally
    ) {
        Spacer(modifier = Modifier.height(16.dp))
        if (selectedVideoSource == VideoSource.Video1) {
            VideoPlayer(
                videoUrl = mainViewModel.currentVideo,
                modifier = Modifier
                    .fillMaxWidth()
                    .aspectRatio(16 / 9f)
            )
        } else {
            VideoPlayer(
                videoUrl = mainViewModel.currentVideo,
                modifier = Modifier
                    .fillMaxWidth()
                    .aspectRatio(16 / 9f)
            )
        }
        LazyColumn(
            modifier = Modifier
                .fillMaxWidth()
                .weight(1f)
        ) {
            items(mainViewModel.channels) { channel ->
                Row(
                    modifier = Modifier
                        .fillMaxWidth()
                        .padding(16.dp)
                        .clickable {
                            selectedChannel.value = channel
                            mainViewModel.currentVideo = channel.location
                            selectedVideoSource = when (selectedVideoSource) {
                                VideoSource.Video1 -> VideoSource.Video2
                                VideoSource.Video2 -> VideoSource.Video1
                            }
                        },
                    verticalAlignment = Alignment.CenterVertically
                ) {
                    Image(
                        painter = rememberAsyncImagePainter(
                            ImageRequest.Builder(LocalContext.current).data(data = channel.image)
                                .apply(block = fun ImageRequest.Builder.() {
                                    crossfade(true)
                                }).build()
                        ),
                        contentDescription = Constants.IMAGE_CONTENT_DESCRIPTION,
                        modifier = Modifier
                            .size(64.dp)
                            .clip(CircleShape)
                    )
                    Spacer(modifier = Modifier.width(16.dp))
                    Text(
                        text = channel.name,
                        style = MaterialTheme.typography.bodyLarge
                    )
                }
            }
        }
    }
}

fun getChannels(url: String): Array {
    val client = OkHttpClient()
    val request = Request.Builder()
        .url(url)
        .build()

    val response = try {
        client.newCall(request).execute().body?.string()
    } catch (e: Exception) {
        // Handle invalid URL or network error
        // Could use e.printStackTrace()
        null
    }

    val channels = mutableListOf()
    var id = 1
    if (response != null) {
        val lines = response.split(Constants.NEWLINE)
        for (i in lines.indices) {
            if (lines[i].startsWith(Constants.LINE_TAG)) {
                val tvgId = getTvgId(lines[i])
                val tvgLogo = getTvgLogo(lines[i])
                if (i + 1 < lines.size) {
                    val location = lines[i + 1]
                    channels.add(Channel(id, tvgLogo, tvgId, location))
                    id++
                }
            }
        }
    }
    return channels.toTypedArray()
}

private fun getTvgId(line: String): String {
    return extractValue(line, Constants.ID_TAG)
}

private fun extractValue(line: String, tag: String): String {
    val start = line.indexOf(tag) + tag.length + 1
    val end = line.indexOf(Constants.QUOTE, start)
    return line.substring(start, end)
}

private fun getTvgLogo(line: String): String {
    return extractValue(line, Constants.LOGO_TAG)
}

@OptIn(UnstableApi::class)
@Composable
fun VideoPlayer(
    videoUrl: String,
    modifier: Modifier = Modifier
) {
    val context = LocalContext.current
    val exoPlayer = remember {
        ExoPlayer.Builder(context)
            .build()
            .apply {
                setMediaSource(
                    HlsMediaSource.Factory(DefaultHttpDataSource.Factory())
                        .createMediaSource(MediaItem.fromUri(videoUrl))
                )
                prepare()
                play()
            }
    }

    DisposableEffect(exoPlayer) {
        onDispose {
            exoPlayer.release()
        }
    }

    Surface(
        modifier = modifier.fillMaxSize(),
        color = MaterialTheme.colorScheme.background
    ) {
        Box(
            modifier = Modifier.fillMaxSize(),
            contentAlignment = Alignment.Center
        ) {
            AndroidView(
                modifier = Modifier.fillMaxSize(),
                factory = { context ->
                    PlayerView(context).apply {
                        player = exoPlayer
                        useController = true
                        controllerShowTimeoutMs = 5000
                    }
                }
            )
        }
    }
}

/* SettingsScreen */
@Composable
fun SettingsScreen(mainViewModel: MainViewModel) {
    var newPlaylistUrl by remember { mutableStateOf(mainViewModel.currentPlaylist) }
    var newVideoUrl by remember { mutableStateOf(mainViewModel.currentVideo) }
    val focusManager = LocalFocusManager.current

    Column(
        modifier = Modifier
            .fillMaxSize()
            .padding(horizontal = 16.dp),
        verticalArrangement = Arrangement.Top,
        horizontalAlignment = Alignment.CenterHorizontally
    ) {
        Text(
            text = Constants.SETTINGS_TITLE,
            style = MaterialTheme.typography.headlineMedium
        )
        Spacer(modifier = Modifier.height(16.dp))
        TextField(
            value = newPlaylistUrl,
            onValueChange = { newPlaylistUrl = it },
            label = { Text(Constants.SETTINGS_CURRENT_PLAYLIST_ENTRY) },
            modifier = Modifier.fillMaxWidth(),
            keyboardOptions = KeyboardOptions(imeAction = ImeAction.Done),
            keyboardActions = KeyboardActions(
                onDone = {
                    mainViewModel.currentPlaylist = newPlaylistUrl
                    focusManager.clearFocus()
                }
            )
        )
        Spacer(modifier = Modifier.height(16.dp))
        TextField(
            value = newVideoUrl,
            onValueChange = { newVideoUrl = it },
            label = { Text(Constants.SETTINGS_CURRENT_VIDEO_ENTRY) },
            modifier = Modifier.fillMaxWidth(),
            keyboardOptions = KeyboardOptions(imeAction = ImeAction.Done),
            keyboardActions = KeyboardActions(
                onDone = {
                    mainViewModel.currentVideo = newVideoUrl
                    focusManager.clearFocus()
                }
            )
        )
    }
}
